  raise Error::Operation.new [

    ldap.get_operation_result.error_message
  ].compact
end

def net_ldap
  auth = {
    :method => :simple,
    :username => "cn=admin,dc=engines,dc=internal",
    :password => "password"
  }
  Net::LDAP.open(host: "ldap", auth: auth) do |conn|
    yield conn
  end
end
