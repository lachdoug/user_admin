def create_users_account_email( user_uid, email )
  net_ldap do |ldap|
    begin
      raise Error unless create_users_account_email_query( ldap, user_uid, email )
      return email
    rescue Error => e
      ldap_op_error( ldap, "Failed to create account email." )
    end
  end
end
