def self.controller_const_for_file( file )
  const_get(
    file.sub( './v0/api/controllers/', '' ).
    sub( '.rb', '' ).
    split('/').map { |const|
      const.split('_').collect(&:capitalize).join
    }.join('::')
  )
end
